7/10/09	DaCosta Under the Realtime Search, Search , Track, we have changed "MarkUp" due to "XML" to .NET object serialization issue.
                
                WAS
                
                [XmlType(TypeName = "Markup", Namespace = Declarations.SchemaVersion), Serializable]
                [EditorBrowsable(EditorBrowsableState.Advanced)]
                public class Markup
                {

                    [XmlAnyElement]
                    public System.Xml.XmlElement Any;
                }
                
                IS 
                
                [XmlType(TypeName = "Markup", Namespace = Declarations.SchemaVersion), Serializable]
                [EditorBrowsable(EditorBrowsableState.Advanced)]
                public class Markup
                {

                    [XmlAnyElement]
                    public System.Xml.XmlElement[] Any;
                }
                
                Example of how to handle the change
                
                WAS
                /// <summary>
                /// Parses the markup.
                /// </summary>
                /// <param name="paras">The paras.</param>
                /// <returns></returns>
                private static List<MarkupItem> ParseMarkup(XmlNode para)
                {                   
                    List<MarkupItem> items = new List<MarkupItem>();
                    if (para == null)
                        return items;
                    foreach (XmlNode node in para.ChildNodes)
                    {
                        MarkupItem item = new MarkupItem();
                        switch (node.NodeType)
                        {
                            case XmlNodeType.Element:
                                switch (node.Name)
                                {
                                    case "hlt":
                                        item.type = EntityType.Highlight.ToString();
                                        item.value = node.InnerText;
                                        items.Add(item);
                                        break;
                                    case "en":
                                        item.type = MapCatToEntityType(node.Attributes.GetNamedItem("cat").Value).ToString();
                                        item.guid = node.Attributes.GetNamedItem("ref").Value;
                                        item.value = node.InnerText;
                                        items.Add(item);
                                        break;
                                }
                                break;
                            case XmlNodeType.Text:
                                item.type = EntityType.Textual.ToString();
                                item.value = node.InnerText;
                                items.Add(item);
                                break;
                        }
                        return items;
                    }
                    return null;
                }

                
                IS
                /// <summary>
                /// Parses the markup.
                /// </summary>
                /// <param name="paras">The paras.</param>
                /// <returns></returns>
                private static List<MarkupItem> ParseMarkup(IEnumerable<XmlNode> paras)
                {
                    List<MarkupItem> items = new List<MarkupItem>();
                    if (paras == null)
                        return items;
                    foreach(XmlNode para in paras)
                    {
                        if (para == null || para.ChildNodes.Count <= 0)
                            continue;                        
                        foreach (XmlNode node in para.ChildNodes)
                        {
                            MarkupItem item = new MarkupItem();
                            switch (node.NodeType)
                            {
                                case XmlNodeType.Element:
                                    switch (node.Name)
                                    {
                                        case "hlt":
                                            item.type = EntityType.Highlight.ToString();
                                            item.value = node.InnerText;
                                            items.Add(item);
                                            break;
                                        case "en":
                                            item.type = MapCatToEntityType(node.Attributes.GetNamedItem("cat").Value).ToString();
                                            item.guid = node.Attributes.GetNamedItem("ref").Value;
                                            item.value = node.InnerText;
                                            items.Add(item);
                                            break;
                                    }
                                    break;
                                case XmlNodeType.Text:
                                    item.type = EntityType.Textual.ToString();
                                    item.value = node.InnerText;
                                    items.Add(item);
                                    break;
                            }
                        }
                        return items;
                    }
                    return null;
                }
